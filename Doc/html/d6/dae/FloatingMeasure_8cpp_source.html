210 void CFloatingMeasure::Normalize()
211 {
212  // first handle simple case: only one CSimpleMeasure. I.e. when
213  // 1. pMeasureRight == nullptr
214  if( Measure().pMeasureRight == nullptr )
215  {
216  // multiply by "own premeasure" * "SIFactor"
217  dfFloating *= Measure().pMeasureLeft->SIFactor();
218  // add SI offset
219  dfFloating += Measure().pMeasureLeft->SIOffset();
220 
221  // set measure to SI
222  cmMeasure.pMeasureLeft->SetByID(pmIdent,BASE->SIID(cmMeasure.pMeasureLeft->BaseID()));
223 
224  // leave now ... we will not do any complex stuff
225  return;
226  }
227 
228  cmMeasure.Normalize();
229  dfFloating *= cmMeasure.ReleaseExp10AndFactor();
230 }

238 void CFloatingMeasure::ScaleTo(const CComplexMeasure& other)
239 {
240  // first handle simple case: only one CSimpleMeasure. I.e. when
241  // 1. pMeasureRight == nullptr for other and this
242  // 2. pMeasureLeft is compatible with other.pMeasureLeft
243  if( Measure().pMeasureRight == nullptr &&
244  other.pMeasureRight == nullptr &&
245  Measure().pMeasureLeft->Compatible(other.pMeasureLeft)
246  )
247  {
248  // receipe: example m°C --> k°F
249  // 1. Normalization: e.g. m°C --> 0.001°K
250  // 2. to target base measure (no pre measure): 0.001°K --> XXX °F
251  // 3. to target base measure with pre measure; XXX °F / 1000 --> XXX / 1000 * k°F
252 
253  // 1, Normalization:
254  // this to SI : resolves own pre measure factor
255  Normalize();
256 
257  // 2. to target base measure (no pre measure)
258  // invert SI conversion of other: / other.SIFactor and - other.SIOffset
259  dfFloating -= other.pMeasureLeft->SIOffset();
260  dfFloating /= BASE->Factor(other.pMeasureLeft->BaseID());
261 
262  // 3. to target base measure with pre measure
263  // now divide by premeasure of other (e.g. m°C --> k°F:
264  dfFloating /= PRE->Factor(other.pMeasureLeft->PreID());
265 
266  // hand over the new measure
267  cmMeasure = other;
268  }
269  else
270  {
271  // handle complex case:
272  // check compatibility
273  if (!Compatible(other))
274  {
275  // invalidate content
276  _Init();
277  }
278  else
279  {
280  cmMeasure.ScaleTo(other);
281  dfFloating *= cmMeasure.ReleaseExp10AndFactor();
282  }
283  }
284  // scale precision .. if necessary:
285  // REM: this is a recursive call
286  if (pfmPrecision != nullptr)
287  _ScalePrecision(other);
288 }

328 void CFloatingMeasure::_ScalePrecision(const CComplexMeasure& other)
329 {
330  pfmPrecision->ScaleTo(other);
331 
332  // do the following only if the precision has a compatible measure
333  // otherwise CFloatingMeasure::ScaleTo will reset *pfmPrecision to invalid values
334  // see CFloatingMeasure::_Init()
335  if( pfmPrecision->Valid())
336  {
337  // calculate the resolution in digits
338  // provide next higher number of digits for the given resolution (dValueToDeriveDigitsFrom)
339  // nDigit = 1 for resolution in [0.1, 1[
340  // nDigit = 0 for resolution in [1, 10[
341  // nDigit = -1 for resolution in [10, 100[
342  CDigFloat dfDigits = log(pfmPrecision->Floating(), 10)*(-1.);
343 
344  // for positive number or zero digits (i.e. precision < 10) add a digit
345  // for negative number of digits subtracting twice the error should be sufficient
346  // to achieve the desired number of digits
347  if (pfmPrecision->Floating() < 10)
348  dfDigits += 1.;
349 
350  // subtract 2* the error to be sure to fall below the next integer before casting to integer
351  // REM: if you subtract only once it might happen that the subtracted result will not be different from the original double.
352  // This is due to numerical errors when using floating numbers. Subtracting twice the error will make
353  // sure to hop over the gap.
354  int nDigits = (int)(dfDigits.RawValue()-2*dfDigits.RawError());
355 
356  // set to new precision
357  dfFloating.Precision(nDigits);
358 
359  // activate precision
360  PrecisionActive(true);
361  }
362 }
