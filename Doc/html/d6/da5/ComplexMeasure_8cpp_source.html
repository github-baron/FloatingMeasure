130 bool CComplexMeasure::operator==(const CComplexMeasure& other) const
131 {
132  // equality is achieved if meaure1 / measure2 = 1
133  // make all calculations with the precision of this
134  CComplexMeasure Check(*this);
135  Check /= other;
136 
137  // do a simplification
138  Check.Simplify();
139  Check.Normalize();
140  return Check.CMFactor() == 1 &&
141  Check.CMExp10() == 0 &&
142  Check.pMeasureRight == nullptr &&
143  Check.pMeasureLeft->BaseID() == bmNumber;

186 void CComplexMeasure::_AllocateLastUnsetMeasureRight(CComplexMeasure& cmOperandLeft, const CComplexMeasure& cmOperandRight, const eOperation OperationEnum)
187 {
188  // go to the end of the line and add the new complex measure with multiplication
189  CComplexMeasure* pCMLeftLast = cmOperandLeft.pMeasureRight;
190  if( pCMLeftLast == nullptr )
191  pCMLeftLast = &cmOperandLeft;
192 
193  // search for last set pMeasureRight: pCMLeftLast is set to this pointer
194  while(pCMLeftLast->pMeasureRight != nullptr )
195  pCMLeftLast = pCMLeftLast->pMeasureRight;
196 
197  // add the new complex number at the end ... and the corresponding operator
198  pCMLeftLast->pMeasureRight = new CComplexMeasure(cmOperandRight);
199  pCMLeftLast->opEnum = OperationEnum;
200  if(OperationEnum == opDivide)
201  pCMLeftLast->pMeasureRight->_InvertOP();
202 
203  // in case the cmOperandRight has factors and exp10 :
204  // collect the nCMExp10 and dfCMFactor to cmOperandLeft::nCMExp10 and cmOperandLeft::dfCMFactor
205  cmOperandLeft._COllectCMExp10CMFactorWrapper();
206 }

250 void CComplexMeasure::_Normalize(eOperation* &pPreviousOP, int* &pdTempPreMeasureExp10Collector , CDigFloat* &pdTempCMFactorCollector, CComplexMeasure* &pFirstComplexMeasure)
251 {
252  // set the collector pointers if not set :
253  // this happens when the CComplexMeasure::_Normalize is called the first time
254  if( pPreviousOP == nullptr){
255 
256  // init all factor collectors with 1
257  pdTempPreMeasureExp10Collector = new int[1];
258  pdTempCMFactorCollector = new CDigFloat(1);
259  pdTempPreMeasureExp10Collector[0] = 0;
260 
261  // first operand is always on the multiplication side
262  pPreviousOP = new eOperation[1];
263  pPreviousOP[0] = (opMult);
264 
265  // ... and for the final results: these will be set at the end of the consecutive recursive calls
266  pFirstComplexMeasure = this;
267 
268  }
269 
270  // set the pre factor and SI factor applying the corresponding operator
271  switch(pPreviousOP[0])
272  {
273  case opMult:
274 
275  // pre measures can be added any time: no offsets to "fear" --> set PreMeaures to identity
276  pdTempPreMeasureExp10Collector[0] += PRE->Exp10( pMeasureLeft->PreID() ) + CMExp10();
277 
278  // CMExp10 can be set to identity right now
279  CMExp10(0);
280 
281  // get CMFactor, too
282  *pdTempCMFactorCollector *= CMFactor();
283 
284  // reset the CMFactor to 1
285  CMFactor(1);
286 
287  // base units with offset will be neglected --> this is trouble in complex measures
288  if( BASE->Offset( pMeasureLeft->BaseID() ) == 0 )
289  {
290  // get the base SIfactor
291  *pdTempCMFactorCollector *= BASE->Factor( pMeasureLeft->BaseID() );
292 
293  // reset pMeasureLeft to SI
294  pMeasureLeft->SetByID(pmIdent, BASE->SIID(pMeasureLeft->BaseID()) );
295  }
296  else
297  {
298  // reset only the PreMeasurefactor of pMeasureLeft to identity
299  pMeasureLeft->SetByID(pmIdent, pMeasureLeft->BaseID());
300  }
301 
302  break;
303 
304  case opDivide:
305 
306  // pre measures can be added any time: no offsets to "fear"
307  pdTempPreMeasureExp10Collector[0] -= PRE->Exp10( pMeasureLeft->PreID() ) + nCMExp10;
308 
309  // CMExp10 can be set to 0 right now
310  CMExp10(0);
311 
312  // get CMFactor, too
313  *pdTempCMFactorCollector /= CMFactor();
314 
315  // reset the CMFactor to 1
316  CMFactor(1);
317 
318  // base units with offset will be neglected --> this is trouble in complex measures
319  if( BASE->Offset( pMeasureLeft->BaseID() ) == 0 )
320  {
321  // get the base SIfactor
322  *pdTempCMFactorCollector /= BASE->Factor( pMeasureLeft->BaseID() );
323 
324  // reset pMeasureLeft to SI
325  pMeasureLeft->SetByID(pmIdent, BASE->SIID(pMeasureLeft->BaseID()) );
326  }
327  else
328  {
329  // reset only the PreMeasurefactor of pMeasureLeft to identity
330  pMeasureLeft->SetByID(pmIdent, pMeasureLeft->BaseID());
331  }
332 
333  break;
334 
335  default:
336 
337  break;
338 
339  } // endswitch(actOP)
340 
341 
342  // if pMeasureRight is a nullptr --> recursion stops here:
343  // finally calculate dfCMFactor and nCMExp10
344  if( pMeasureRight == nullptr )
345  {
346 
347  // setting the factor completely new
348  pFirstComplexMeasure->dfCMFactor = *pdTempCMFactorCollector;
349 
350  // subtract the first exponent ... otherwise it will be counted twice
351  pFirstComplexMeasure->CMExp10(pdTempPreMeasureExp10Collector[0]-pFirstComplexMeasure->CMExp10());
352 
353  // tidy up
354  SecureDeleteObjectPointer(pdTempPreMeasureExp10Collector);
355  SecureDeleteObjectPointer(pdTempCMFactorCollector);
356  SecureDeleteVectorPointer(pPreviousOP);
357 
358  } // endif( pMeasureRight == nullptr )
359  else
360  {
361  // ... otherwise go on:
362  // set operator correctly
363  pPreviousOP[0] = opEnum;
364 
365  // and call recursively
366  pMeasureRight->_Normalize(pPreviousOP, pdTempPreMeasureExp10Collector, pdTempCMFactorCollector, pFirstComplexMeasure );
367 
368  }
369

386 void CComplexMeasure::_COllectCMExp10CMFactor(eOperation *& pPreviousOP, int *& pdTempPreMeasureExp10Collector, CDigFloat*& pdTempCMFactorCollector, CComplexMeasure* &pFirstComplexMeasure)
387 {
388 
389  // set the collector pointers if not set :
390  // this happens when the primary CComplexMeasure::Normalize is called
391  if( pPreviousOP == nullptr){
392 
393  // init all factor collectors with 1
394  pdTempPreMeasureExp10Collector = new int[1];
395  pdTempCMFactorCollector = new CDigFloat(1);
396  pdTempPreMeasureExp10Collector[0] = 0;
397 
398 
399  // first operand is always mult
400  pPreviousOP = new eOperation[1];
401  pPreviousOP[0] = (opMult);
402 
403  // ... and for the final results: these will be set at the end of the consecutive recursive calls
404 // pnCMExp10 = &nCMExp10;
405 // pdfCMFactor = &dfCMFactor;
406  pFirstComplexMeasure = this;
407 
408  }
409 
410  // set the pre factor and SI factor applying the corresponding operator
411  switch(pPreviousOP[0])
412  {
413  case opMult:
414 
415  // CM pre measures exponents can be added
416  pdTempPreMeasureExp10Collector[0] += CMExp10();
417 
418  // reset to identity for this
419  CMExp10(0);
420 
421  // get the base SIfactor
422  *pdTempCMFactorCollector *= CMFactor();
423 
424  //reset the CMFactor for this
425  CMFactor(1);
426 
427  break;
428 
429  case opDivide:
430 
431 
432  // CM pre measures exponents can be added
433  pdTempPreMeasureExp10Collector[0] -= CMExp10();
434 
435  // reset to identity
436  CMExp10(0);
437 
438  // get the base SIfactor
439  *pdTempCMFactorCollector /= CMFactor();
440 
441  //reset the CMFactor for this
442  CMFactor(1);
443 
444  break;
445 
446  default:
447 
448  break;
449 
450  } // endswitch(actOP)
451 
452 
453  // if pMeasureRight is a nullptr --> recursion stops here:
454  // finally calculate dSIFactor and eSIPreMeasure
455  if( pMeasureRight == nullptr )
456  {
457 
458  pFirstComplexMeasure->dfCMFactor = *pdTempCMFactorCollector;
459  pFirstComplexMeasure->CMExp10(pdTempPreMeasureExp10Collector[0]);
460 
461  // tidy up
462  SecureDeleteObjectPointer(pdTempPreMeasureExp10Collector);
463  SecureDeleteObjectPointer(pdTempCMFactorCollector);
464  SecureDeleteVectorPointer(pPreviousOP);
465 
466  } // endif( pMeasureRight == nullptr )
467  else
468  {
469  // ... otherwise go on:
470  // set operator correctly
471  pPreviousOP[0] = opEnum;
472 
473  // and call recursively
474  pMeasureRight->_COllectCMExp10CMFactor(pPreviousOP,pdTempPreMeasureExp10Collector , pdTempCMFactorCollector, pFirstComplexMeasure);
475 
476  }

480 void CComplexMeasure::Simplify()
481 {
482  // search for the next CSimpleMeasure in the chain with the same SI unit index
483  // hand over the starting pointer CComplexMeasure*, the corresponding CMExp10 and CMFactor pointers
484  // which will be filled
485  eOperation eFirstOP = opMult;
486  _Simplify(eFirstOP);
487 
488  // at this point tidy up:
489  // collect all CMExp10 and CMFactor
490  _COllectCMExp10CMFactorWrapper();
491 
492  // clear all measureless neutral elements (pmIdent, bmNumber)(except at the very start)
493  // generated by _SearchAndShortenSameSIIndex
494  if(pMeasureRight != nullptr)
495  pMeasureRight->_RemoveNeutralNumbers(*this);
496 
497  // in case the operator is opMult and the first measure is a bmNumber --> remove this pointer, too
498  // replace it by the pMeasureRight: e.g. 1*s --> s
499  if(opEnum == opMult && pMeasureLeft->BaseID() == bmNumber && pMeasureRight != nullptr)
500  {
501  // 1. set the pMeasureLeft to pMeasureLeft of the consecutive ComplexMeasure
502  // 2. set the op to op of the consecutive ComplexMeasure
503  pMeasureLeft->operator=(pMeasureRight->pMeasureLeft);
504  opEnum = pMeasureRight->opEnum;
505 
506  // 3. tidy up pMeasureLeft and set op of consecutive ComplexMeasure
507  SecureDeleteObjectPointer(pMeasureRight->pMeasureLeft);
508  pMeasureRight->opEnum = opUnknown;
509 
510  // 4. set pMeasureRight of this to pMeasureRight of consecutive ComplexMeasure
511  pMeasureRight = pMeasureRight->pMeasureRight;
512 
513  }

516 void CComplexMeasure::_Simplify(eOperation& PreviousOP)
517 {
518 
519  // search for a SI partner with inverted OP for cancellation
520  // hand over
521  // 1. the pointer of calling instance
522  // 2. the operator of this calling instance
523  // 3. the operator preceeding pMeasureRight
524  // for the first call the calling instance is this
525  // and the operator of the preceeding pMeasureRight is the opEnum of this
526  if( pMeasureRight != nullptr)
527  pMeasureRight->_SearchAndShortenSameSIIndex(this, PreviousOP, opEnum);
528 
529  // recursive call of all consecutive ComplexMeasures which search
530  // for SI partners and cancel with them setting both to measureless number
531  if(pMeasureRight != nullptr)
532  pMeasureRight->_Simplify(opEnum);
533 }

536 void CComplexMeasure::_RemoveNeutralNumbers(CComplexMeasure &PreviousComplexMeasure)
537 {
538  // this takes us to the end of the ComplexMeasure list
539  if(pMeasureRight != nullptr)
540  pMeasureRight->_RemoveNeutralNumbers(*this);
541 
542  // if we come here : the end was reached and we are going backward in the list
543  // check if this simple measure is measureless
544  if( pMeasureLeft->PreID() == pmIdent && pMeasureLeft->BaseID() == bmNumber)
545  {
546  // set the reference here to the successor ... or nullptr
547  PreviousComplexMeasure.pMeasureRight = pMeasureRight;
548 
549  // in case the successor is null --> opEnum = opUnknown
550  if(pMeasureRight == nullptr)
551  PreviousComplexMeasure.opEnum = opUnknown;
552 
553  // delete pMeasureLeft, reset pMeasureRight
554  pMeasureRight = nullptr;
555  opEnum = opUnknown;
556  SecureDeleteObjectPointer(pMeasureLeft);
557  }

561 void CComplexMeasure::_SearchAndShortenSameSIIndex(CComplexMeasure* pShortenPartner,eOperation& eOPofShortenPartner, eOperation& eOPofConsecutiveShortenPartner)
562 {
563 
564  // in case SI Index of simple measure is not the same
565  // or
566  // one of the partnes is already cancelled (i.e. BaseMeasureEnum == bmNumber)
567  // or
568  // the offsets for recalculation to SI are not the same (mathematical desaster in general which I will not solve here)
569  // or
570  // the operators are the same
571  // --> go on
572  if( BASE->SIID( pShortenPartner->pMeasureLeft->BaseID() ) != BASE->SIID( pMeasureLeft->BaseID()) ||
573  BASE->SIID( pShortenPartner->pMeasureLeft->BaseID() ) == bmNumber ||
574  BASE->SIID( pMeasureLeft->BaseID()) == bmNumber ||
575  pShortenPartner->pMeasureLeft->SIOffset() != pMeasureLeft->SIOffset() ||
576  eOPofShortenPartner == eOPofConsecutiveShortenPartner)
577  {
578  // recursive calll ... only if there is a successor
579  if(pMeasureRight != nullptr)
580  pMeasureRight->_SearchAndShortenSameSIIndex(pShortenPartner, eOPofShortenPartner, opEnum);
581  }
582  else
583  {
584  pShortenPartner->_ResolveMeasureLeft();
585  _ResolveMeasureLeft();
586  }
