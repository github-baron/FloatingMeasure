142 ePreMeasure CPreMeasure::GetIDByFactor(const double dPreMeasureFactor)
143 {
144  // handle extreme cases:
145  //dPreMeasureFactor smaller equal than smallest value of PreMeasure factors
146  if(dPreMeasureFactor <= Factor(pmFirst))
147  return pmFirst;
148 
149  //dPreMeasureFactor bigger equal than biggest value of PreMeasure factors
150  if(dPreMeasureFactor >= Factor(pmLast-1))
151  return (ePreMeasure)(pmLast-1);
152 
153  // if we come here iterate over all other PreMeasure factors:
154  // stop in case the remaining factor is between 0.1 and 10
155  ePreMeasure pmOpt = pmUnknown;
156  for(unsigned int idx = 0; idx < vdFactor->size(); idx++)
157  {
158  // set the limits depending of the consecutive and previous measure:
159  // default is a factor 1000 distance:
160  // there are only factors of 1000 or 10 (ideally)
161  double dLowerLimit = 0.099999999; //ideally 0.1
162  double dUpperLimit = 10.000000001; //ideally 10
163 
164  if(idx>0)
165  if( (Factor(idx)/Factor(idx-1)) < 100)
166  dLowerLimit = 0.9;
167 
168  if((Factor(idx+1)/Factor(idx)) < 100)
169  dUpperLimit = 1.1;
170 
171  double dRatio = dPreMeasureFactor/Factor(idx);
172  if( (dLowerLimit < dRatio) && (dRatio < dUpperLimit))
173  {
174  pmOpt = (ePreMeasure)(idx);
175  break;
176  }
177 
178  }
179 
180  return pmOpt;
181 }
